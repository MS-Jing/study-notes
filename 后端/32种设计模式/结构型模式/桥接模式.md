拆分两端，解耦合。

```java
package com.lj.bridge;

/**
 * 品牌
 * @author luojing
 * @create 2021/4/6
 */
public interface Brand {
    void info();
}

```

```java
/**
 * @author luojing
 * @create 2021/4/6
 */
public class Apple implements Brand{
    @Override
    public void info() {
        System.out.println("苹果品牌");
    }
}
```

```java
/**
 * @author luojing
 * @create 2021/4/6
 */
public class Computer {
    //组合品牌
    protected Brand brand;

    public Computer(Brand brand) {
        this.brand = brand;
    }

    public void info(){
        brand.info();
    }
}

class DeskTop extends Computer{

    public DeskTop(Brand brand) {
        super(brand);
    }

    @Override
    public void info() {
        super.info();
        System.out.println("台式机");
    }
}

class NoteBook extends Computer{

    public NoteBook(Brand brand) {
        super(brand);
    }

    @Override
    public void info() {
        super.info();
        System.out.println("笔记本");
    }
}
```

```java
/**
 * @author luojing
 * @create 2021/4/6
 */
public class Main {
    public static void main(String[] args) {
        DeskTop deskTop = new DeskTop(new Apple());
        deskTop.info();
    }
}
```
